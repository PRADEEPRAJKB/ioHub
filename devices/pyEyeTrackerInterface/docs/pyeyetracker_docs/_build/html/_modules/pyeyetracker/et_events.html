from et_constants import *
from collections import namedtuple

UserEvent = namedtuple('UserEvent','localTime eventType namedtuplePayload',verbose=True)

KeyboardEvent = namedtuple('KeyboardEvent','localTime key')
MouseButtonEvent = namedtuple('MouseButtonEvent','localTime button')

TrackerEvent = namedtuple('TrackerEvent','localTime trackerTime')
EyeEvent = namedtuple('EyeEvent',TrackerEvent._fields+('eye',))

Sample = namedtuple('Sample','gaze pupilSize ppd status extendedData')
SampleExtendedData = namedtuple('SampleExtended','angle pupil cr crSize headData input button')#, verbose=True)

MonocularSample = namedtuple('MonocularSample',EyeEvent._fields+Sample._fields)#, verbose=True)
BinocularSample = namedtuple('BinocularSample',EyeEvent._fields+('leftEye','rightEye'))#, verbose=True)

MessageEvent = namedtuple('Message',TrackerEvent._fields+('text',))#, verbose=True)
DigitalInputEvent = namedtuple('DigitalInput',TrackerEvent._fields+('value',))#, verbose=True)
ButtonEvent = namedtuple('ButtonEvent',TrackerEvent._fields+('buttons_pressed','buttons_released'))#, verbose=True)

FixationStartEvent = namedtuple('FixationStartEvent',EyeEvent._fields+('startGazePosition','startPupilSize','startPPD','startVelocity'))
FixationEndEvent = namedtuple('FixationEndEvent',EyeEvent._fields+('endGazePosition','endPupilSize','endPPD','endVelocity'))
FixationEvent = namedtuple('FixationEvent',('startEvent','endEvent','duration','averageGazePosition','averagePupilSize','averagePPD','averageVelocity','peakVelocity'))

SaccadeStartEvent = namedtuple('SaccadeStartEvent',EyeEvent._fields+('startGazePosition','startPupilSize','startPPD','startVelocity'))
SaccadeEndEvent = namedtuple('SaccadeEndEvent',EyeEvent._fields+('endGazePosition','endPupilSize','endPPD','endVelocity'))
SaccadeEvent = namedtuple('SaccadeEvent',('startEvent','endEvent','duration','amplitude','averageGazePosition','averagePupilSize','averagePPD','averageVelocity','peakVelocity'))

FixatonUpdateEvent = namedtuple('FixatonUpdateEvent',('startEvent','endEvent','duration','averageGazePosition','averagePupilSize','averagePPD','averageVelocity','peakVelocity'))
  
BlinkStartEvent = namedtuple('BlinkStartEvent',EyeEvent._fields)
BlinkEndEvent = namedtuple('BlinkEndEvent',EyeEvent._fields)
BlinkEvent = namedtuple('BlinkEvent',('startEvent','endEvent','duration'))
